export class AnthropicStyle implements ProviderStyle {
  readonly id = 'anthropic' as const;
  constructor(private apiKey: string, private baseUrl = 'https://api.anthropic.com/v1') {}

  async *stream(req: ChatRequest): AsyncIterable<ChatChunk> {
    const res = await fetch(`${this.baseUrl}/messages`, {
      method: 'POST',
      signal: req.signal,
      headers: {
        'content-type': 'application/json',
        'x-api-key': this.apiKey,
        'anthropic-version': '2023-06-01',

      body: JSON.stringify({ model: req.model, stream: true, messages: req.messages, tools: req.tools }),
    });
    if (!res.ok || !res.body) throw new ProviderError(res.status, await res.text());

    const stream = res.body.pipeThrough(new TextDecoderStream()).pipeThrough(createParser());
    const partialArgs = new Map<string, string>();

    for await (const event of stream as unknown as AsyncIterable<EventSourceMessage>) {
      if (event.event === 'content_block_delta') {
        const d = JSON.parse(event.data) as AnthropicDelta;
        if (d.delta.type === 'text_delta') yield { kind: 'text', text: d.delta.text };
        if (d.delta.type === 'input_json_delta') {
          const id = String(d.index);
          partialArgs.set(id, (partialArgs.get(id) ?? '') + d.delta.partial_json);
          yield { kind: 'tool_call_delta', toolCallId: id, argDelta: d.delta.partial_json };
        }
      }
      if (event.event === 'message_stop') yield { kind: 'done' };
    }
  }
}
