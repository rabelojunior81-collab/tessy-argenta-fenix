// Import the new AI system
import { ProviderRegistry } from '../lib/ai/providers/ProviderRegistry';
import { ToolCallDispatcher } from '../lib/ai/tools/ToolCallDispatcher';
import { createToolBroker } from '../lib/ai/index';
import { TokenCounter } from '../lib/ai/context/TokenCounter';
import { SlidingWindow } from '../lib/ai/context/SlidingWindow';
import type { ChatMessage } from '../lib/ai/providers/ProviderStyle';

// In the component that uses ChatContext:
const sendMessage = useCallback(async (content: string, attachments?: File[]) => {
  if (!content.trim() && (!attachments || attachments.length === 0)) return;

  const abortController = new AbortController();

  // 1. Build message array (with context window compaction)
  const messages = build messages from conversation history;
  const model = getModelForProvider(provider);
  
  // Compact if needed (SlidingWindow)
  const compacted = await slidingWindow.compact(messages, model, tokenCounter, (reason) => {
    // Could show a toast: "Context compacted"
  });

  // 2. Get provider instance
  const providerInstance = await ProviderRegistry.get(secrets).get(provider);
  
  // 3. Create tool broker with workspace sandbox
  const broker = createToolBroker(secrets);
  const sandbox = new WorkspaceSandbox(workspaceHandle, sessionToken);
  
  // 4. Stream with tool handling
  const dispatcher = new ToolCallDispatcher();
  
  // Add user message to conversation immediately (optimistic)
  addMessage({ role: 'user', content });
  
  // Add assistant message placeholder
  const assistantMsgId = addMessage({ role: 'assistant', content: '' });
  
  // Start streaming
  setIsStreaming(true);
  setStreamingText('');
  
  try {
    const stream = await providerInstance.stream({
      model,
      messages: compacted,
      tools: registeredTools,
      signal: abortController.signal,
    });
    
    // Process stream with tool call handling
    await dispatcher.consume(
      stream,
      // onText: accumulate and display
      (text) => {
        setStreamingText(prev => prev + text);
        updateMessageContent(assistantMsgId, prev => prev + text);
      },
      // onToolCall: show tool call card
      (toolCall) => {
        addToolCallCard(toolCall);
      }
    );
    
    // Execute approved tool calls
    const dispatchResult = dispatcher.getResult();
    
    for (const toolCall of dispatchResult.toolCalls) {
      if (toolCall.status === 'approved') {
        // Execute tool
        const result = await broker.invoke(
          toolCall.toolName,
          toolCall.argsBuffer, // parsed args
          { sandbox, ui: toolCallUI, signal: abortController.signal }
        );
        
        // Re-inject result as tool message
        addMessage({
          role: 'tool',
          content: result.content,
          toolCallId: toolCall.toolCallId,
          toolName: toolCall.toolName,
          toolResult: result.data,
        });
        
        // Continue stream with tool result
        // (Depending on provider, this may require a new stream call or be handled in the same stream)
      }
    }
    
  } catch (err) {
    if (err instanceof Error && err.name === 'AbortError') {
      // User cancelled
    } else {
      // Show error in chat
      addMessage({ role: 'assistant', content: `Error: ${err}` });
    }
  } finally {
    setIsStreaming(false);
    setStreamingText('');
    updateTokenBudget();
  }
}, [provider, conversation, workspaceHandle, sessionToken]);
```

// Tool execution endpoint
app.post('/ai/tools/execute', async (req, res) => {
  const { toolName, args, sessionToken } = req.body;
  
  // Validate session token (existing pattern from server/index.ts L84-98)
  const session = validateSessionToken(sessionToken);
  if (!session) {
    return res.status(401).json({ error: 'Invalid session' });
  }
  
  // Route to appropriate handler
  if (toolName === 'runCommand') {
    // Uses existing PTY broker pattern
    const { command, cwd } = args;
    const result = await executeInPTY(command, cwd ?? session.cwd);
    res.json(result);
  } else {
    // Other tools don't need server proxy — they use browser APIs
    res.status(400).json({ error: `Tool ${toolName} does not need server proxy` });
  }
});

// Streaming proxy (for cases where browser can't call provider directly)
app.post('/ai/stream', async (req, res) => {
  const { provider, model, messages, apiKey } = req.body;
  
  // Validate session
  const session = validateSessionToken(req.headers['x-session-token'] as string);
  if (!session) {
    return res.status(401).json({ error: 'Invalid session' });
  }
  
  // Get provider URL and delegate
  const { url, headers } = getProviderConfig(provider, apiKey);
  
  // Proxy the streaming request
  const response = await fetch(url, {
    method: 'POST',
    headers: { ...headers, 'Content-Type': 'application/json' },
    body: JSON.stringify({ model, messages, stream: true }),
  });
  
  // Pipe SSE stream to client
  res.setHeader('Content-Type', 'text/event-stream');
  response.body?.pipe(res);
});
```

**IMPORTANT:** The existing server/index.ts has PTY broker logic. Use the same session token pattern and extend with AI tool endpoints. Do NOT rewrite the existing server — only add new endpoints.
