```typescript
function ProviderKeyInput({ provider }: { provider: AuthProvider }) {
  const [value, setValue] = useState('');
  const [status, setStatus] = useState<'empty' | 'valid' | 'invalid'>('empty');
  
  useEffect(() => {
    getToken(provider.id).then(token => {
      if (token) {
        setValue(token);
        setStatus(provider.validator(token) ? 'valid' : 'invalid');
      }
    });
  }, [provider.id]);
  
  const handleChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    const newValue = e.target.value;
    setValue(newValue);
    setStatus(newValue === '' ? 'empty' : provider.validator(newValue) ? 'valid' : 'invalid');
    setToken(provider.id, newValue);
  };
  
  const Icon = provider.icon;
  
  return (
    <div className="flex items-center gap-2 py-1">
      <Icon size={14} style={{ color: provider.color }} />
      <div className="flex-1">
        <div className="flex items-center justify-between">
          <span className="text-[10px] font-medium text-glass">{provider.name}</span>
          {status === 'valid' && <span className="text-[8px] text-green-400">✓</span>}
          {status === 'invalid' && <span className="text-[8px] text-red-400">✗</span>}
        </div>
        <input
          type="password"
          value={value}
          onChange={handleChange}
          placeholder={provider.placeholder}
          className="w-full bg-transparent border-b border-glass-border text-[10px] text-glass py-0.5 focus:border-glass-accent focus:outline-none transition-colors"
        />
      </div>
      <a
        href={provider.helpUrl}
        target="_blank"
        rel="noopener noreferrer"
        className="text-[8px] text-glass-muted hover:text-glass-accent transition-colors"
        title={provider.helpText}
      >
        ?
      </a>
    </div>
  );
}
```
