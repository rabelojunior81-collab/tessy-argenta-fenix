import { z } from 'zod';

export interface ToolSpec<TArgs = unknown, TResult = unknown> {
  name: string;
  description: string;
  schema: z.ZodType<TArgs>;
  invoke(
    args: TArgs,
    ctx: {
      sandbox: WorkspaceSandbox;
      ui: ToolUI;
      signal: AbortSignal;
    }
  ): Promise<ToolResult<TResult>>;
}

export interface ToolResult<T = unknown> {
  isError: boolean;
  content: string;
  data?: T;
}

export interface ToolUI {
  requestDiffApproval(ctx: {
    path: string;
    before: string;
    after: string;
  }): Promise<boolean>;
  showProgress?(message: string): void;
}

// Error codes for tool failures
export const ToolErrorCode = {
  E_PATH: 'E_PATH',           // Path traversal or sandbox escape
  E_ARGS: 'E_ARGS',           // Invalid arguments (Zod validation failed)
  E_TOOL_ARGS_RETRY: 'E_TOOL_ARGS_RETRY', // Retryable argument error
  E_INTERNAL: 'E_INTERNAL',   // Internal error during execution
  E_DENYLIST: 'E_DENYLIST',   // Command blocked by denylist
  E_TIMEOUT: 'E_TIMEOUT',     // Execution timed out
} as const;

export type ToolErrorCode = typeof ToolErrorCode[keyof typeof ToolErrorCode];

export class ToolError extends Error {
  constructor(
    public readonly code: ToolErrorCode,
    message: string
  ) {
    super(message);
    this.name = 'ToolError';
  }
}

// Zod schema helpers for common patterns
export const PathSchema = z.string()
  .min(1, 'Path cannot be empty')
  .refine(p => !p.includes('\0'), 'Null-byte prohibited');

export const SafeFilenameSchema = z.string()
  .min(1)
  .refine(p => !p.includes('/') && !p.includes('\\'), 'Must be a filename, not a path');

```typescript
import { ToolError } from './ToolSpec';

export class WorkspaceSandbox {
  constructor(
    private handle: FileSystemDirectoryHandle,
    private sessionToken: string // for delegating to broker when needed
  ) {}

  /**
   * Asserts that the given path is within the workspace root.
   * Uses FSA resolve() as the anchor — this is the ONLY safe way to check containment.
   * 
   * @throws ToolError{E_PATH} if path escapes sandbox
   */
  async assertInside(childPath: string): Promise<void> {
    // 1. Null-byte check
    if (childPath.includes('\0')) {
      throw new ToolError('E_PATH', 'Path contains null-byte');
    }

    // 2. Normalize path
    const normalized = childPath.replace(/\\/g, '/');

    // 3. Use FSA resolve() — this is the anchor of truth for containment
    // If child is outside the handle, resolve() returns null or empty array
    let parts: string[];
    try {
      parts = await this.handle.resolve(normalized);
    } catch {
      throw new ToolError('E_PATH', `Cannot resolve path: ${normalized}`);
    }

    if (!parts || parts.length === 0) {
      throw new ToolError('E_PATH', `Path "${childPath}" is outside workspace`);
    }

    // 4. Additional check: verify the resolved path is within workspaceRoot
    // This guards against symlink-based escapes
    const resolvedParts = [...parts];
    const resolvedPath = `/${resolvedParts.join('/')}`;
    
    // Check for parent directory references that could escape
    if (normalized.includes('..') || normalized.startsWith('/')) {
      // Re-resolve using the handle to get canonical path
      const verifyHandle = await this.handle.getDirectoryHandle(
        resolvedParts.slice(0, -1).join('/'),
        { create: false }
      ).catch(() => null);
      
      if (!verifyHandle) {
        throw new ToolError('E_PATH', 'Path traversal attempt detected');
      }
    }
  }

  /**
   * Read text content from a file within the sandbox
   */
  async readFileText(path: string, encoding: BufferEncoding = 'utf-8'): Promise<string> {
    await this.assertInside(path);
    
    const fileName = path.split('/').pop() ?? path;
    const parentPath = path.slice(0, -fileName.length).replace(/^\//, '');
    
    let parentHandle: FileSystemDirectoryHandle = this.handle;
    if (parentPath) {
      parentHandle = await this.handle.getDirectoryHandle(parentPath, { create: false });
    }
    
    const fileHandle = await parentHandle.getFileHandle(fileName);
    const file = await fileHandle.getFile();
    return new TextDecoder(encoding).decode(await file.arrayBuffer());
  }

  /**
   * Write text content to a file within the sandbox
   */
  async writeFileText(path: string, content: string, encoding: BufferEncoding = 'utf-8'): Promise<void> {
    await this.assertInside(path);
    
    const fileName = path.split('/').pop() ?? path;
    const parentPath = path.slice(0, -fileName.length).replace(/^\//, '');
    
    let parentHandle: FileSystemDirectoryHandle = this.handle;
    if (parentPath) {
      parentHandle = await this.handle.getDirectoryHandle(parentPath, { create: true });
    }
    
    const fileHandle = await parentHandle.createWritable();
    await fileHandle.write(new TextEncoder().encode(content));
    await fileHandle.close();
  }

  /**
   * List directory contents
   */
  async enumerateDir(path: string): Promise<string[]> {
    await this.assertInside(path);
    
    const dirHandle = await this.handle.getDirectoryHandle(path, { create: false });
    const entries: string[] = [];
    for await (const [name] of dirHandle.entries()) {
      entries.push(name);
    }
    return entries;
  }

  get workspaceRoot(): string {
    // FileSystemDirectoryHandle doesn't expose the actual path
    // We track this separately when the workspace is opened
    return 'WORKSPACE_ROOT';
  }

  get sessionToken(): string {
    return this.sessionToken;
  }
}
```

```typescript
import type { ToolSpec, ToolResult } from './ToolSpec';
import { ToolError } from './ToolSpec';
import { WorkspaceSandbox } from './WorkspaceSandbox';

interface SecretStore {
  get(provider: string, key: string): Promise<string | null>;
}

export class ToolBroker {
  private registry = new Map<string, ToolSpec>();
  private secrets: SecretStore;

  constructor(secrets: SecretStore) {
    this.secrets = secrets;
  }

  /**
   * Register a tool with the broker
   */
  register<T extends ToolSpec>(tool: T): void {
    this.registry.set(tool.name, tool);
  }

  /**
   * Invoke a tool by name with given arguments
   */
  async invoke(
    toolName: string,
    rawArgs: unknown,
    ctx: { sandbox: WorkspaceSandbox; ui: ToolUI; signal: AbortSignal }
  ): Promise<ToolResult> {
    const tool = this.registry.get(toolName);
    if (!tool) {
      return { isError: true, content: `Tool "${toolName}" not found.` };
    }

    // 1. Zod validation with 2 retries
    const MAX_RETRIES = 2;
    let parsed = tool.schema.safeParse(rawArgs);
    let retries = 0;

    while (!parsed.success && retries < MAX_RETRIES) {
      const errorMsg = parsed.error.issues
        .map(i => `${i.path.join('.')}: ${i.message}`)
        .join('; ');
      
      retries++;
      if (retries >= MAX_RETRIES) {
        return { isError: true, content: `E_ARGS: ${errorMsg}` };
      }
      
      // Return retryable error — model can fix arguments
      return { isError: true, content: `E_TOOL_ARGS_RETRY: ${errorMsg}` };
    }

    // 2. Hydrate secrets (replace ${PROVIDER.KEY} placeholders)
    const hydratedArgs = this.hydrateSecrets(parsed.data);

    // 3. Execute
    try {
      return await tool.invoke(hydratedArgs, ctx);
    } catch (err) {
      if (err instanceof ToolError) {
        return { isError: true, content: `${err.code}: ${err.message}` };
      }
      return { isError: true, content: `E_INTERNAL: ${(err as Error).message}` };
    }
  }

  /**
   * Hydrate secret placeholders from SecretStore
   * Pattern: ${GITHUB_TOKEN}, ${ANTHROPIC_API_KEY}, etc.
   */
  private hydrateSecrets(args: unknown): unknown {
    return JSON.parse(
      JSON.stringify(args).replace(
        /\$\{(GITHUB|ANTHROPIC|OPENAI|GEMINI|OPENROUTER)_(TOKEN|KEY|SECRET)\}/g,
        (_, provider, key) => {
          const value = this.secrets.get(provider.toLowerCase(), key.toLowerCase());
          return value ?? '';
        }
      )
    );
  }

  /**
   * Get list of registered tool names
   */
  getToolNames(): string[] {
    return Array.from(this.registry.keys());
  }
}
```

**IMPORTANT:** Secret hydration prevents API keys from leaking to the model. The model never sees actual keys — only the hydrated values at invocation time.

```typescript
import type { AsyncIterable } from 'typescript';
import type { ChatChunk } from '../providers/ProviderStyle';
import type { ToolBroker } from './ToolBroker';
import type { WorkspaceSandbox, ToolUI } from './types';

export interface ToolCallState {
  toolCallId: string;
  toolName: string;
  argsBuffer: string;
  status: 'pending' | 'approved' | 'rejected';
}

export interface DispatchResult {
  text: string;
  toolCalls: ToolCallState[];
}

export class ToolCallDispatcher {
  private activeToolCalls = new Map<string, ToolCallState>();

  /**
   * Consume a streaming provider output and collect tool calls
   * Returns accumulated text and tool calls for execution
   */
  async consume(
    stream: AsyncIterable<ChatChunk>,
    onText?: (text: string) => void,
    onToolCall?: (toolCall: ToolCallState) => void
  ): Promise<DispatchResult> {
    let accumulatedText = '';

    for await (const chunk of stream) {
      switch (chunk.kind) {
        case 'text':
          accumulatedText += chunk.text ?? '';
          onText?.(chunk.text ?? '');
          break;

        case 'tool_call_delta':
          // OpenAI/Anthropic: accumulate JSON fragments
          const existing = this.activeToolCalls.get(chunk.toolCallId!);
          if (existing) {
            existing.argsBuffer += chunk.argDelta ?? '';
          } else {
            // New tool call starting
            this.activeToolCalls.set(chunk.toolCallId!, {
              toolCallId: chunk.toolCallId!,
              toolName: chunk.toolName ?? '',
              argsBuffer: chunk.argDelta ?? '',
              status: 'pending',
            });
          }
          onToolCall?.(this.activeToolCalls.get(chunk.toolCallId!)!);
          break;

        case 'tool_call_complete':
          // Gemini/Ollama: complete object received
          const toolCallId = chunk.toolCallId ?? chunk.toolName!;
          this.activeToolCalls.set(toolCallId, {
            toolCallId,
            toolName: chunk.toolName ?? '',
            argsBuffer: typeof chunk.args === 'string' ? chunk.args : JSON.stringify(chunk.args),
            status: 'pending',
          });
          onToolCall?.(this.activeToolCalls.get(toolCallId)!);
          break;

        case 'done':
        case 'usage':
          // Stream complete
          break;
      }
    }

    // Parse tool call arguments from accumulated JSON buffers
    const parsedToolCalls: ToolCallState[] = [];
    for (const [toolCallId, state] of this.activeToolCalls) {
      try {
        const args = JSON.parse(state.argsBuffer);
        parsedToolCalls.push({ ...state, argsBuffer: args as unknown as string });
      } catch {
        // Invalid JSON — mark as rejected
        parsedToolCalls.push({ ...state, status: 'rejected' });
      }
    }

    this.activeToolCalls.clear();

    return {
      text: accumulatedText,
      toolCalls: parsedToolCalls,
    };
  }
}
```

**IMPORTANT:** Handle BOTH `tool_call_delta` (OpenAI/Anthropic — fragments to concatenate) AND `tool_call_complete` (Gemini/Ollama — full object). The dispatcher must handle both patterns since different providers use different strategies.

import { z } from 'zod';
import type { ToolSpec } from '../ToolSpec';
import { ToolError } from '../ToolSpec';

const DENYLIST = [
  /^rm\s+-rf\b/,
  /^git\s+push\s+--force$/,
  /^git\s+push\s+-f$/,
  /:\)\{:\|:&\};/,
  /^DROP\s+DATABASE/i,
  /^DROP\s+TABLE/i,
  /^curl\s+.+\|\s*sh$/,
  /^wget\s+.+\|\s*sh$/,
];

export const RunCommandArgs = z.object({
  command: z.string().min(1),
  cwd: z.string().optional(),
});

export const runCommandTool: ToolSpec<z.infer<typeof RunCommandArgs>> = {
  name: 'runCommand',
  description: 'Execute a shell command in the workspace. Destructive commands require user confirmation.',
  schema: RunCommandArgs,
  async invoke(args, { sandbox, ui, signal }) {
    const cwd = args.cwd ?? sandbox.workspaceRoot;
    
    // Denylist check
    const isDenied = DENYLIST.some(r => r.test(args.command));
    if (isDenied) {
      const confirmed = await ui.requestDestructiveConfirmation?.({
        action: 'runCommand',
        command: args.command,
        reason: 'Potentially destructive command detected.',
      }) ?? false;
      
      if (!confirmed) {
        throw new ToolError('E_DENYLIST', 'Command blocked by denylist');
      }
    }
    
    // Execute via broker (server/index.ts proxy)
    const response = await fetch('http://127.0.0.1:3002/ai/tools/run-command', {
      method: 'POST',
      signal,
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        sessionToken: sandbox.sessionToken,
        command: args.command,
        cwd,
      }),
    });
    
    if (!response.ok) {
      const error = await response.text();
      return { isError: true, content: `Error: ${error}` };
    }
    
    const result = await response.json();
    return {
      isError: false,
      content: result.output ?? result.error ?? 'Done.',
      data: { exitCode: result.exitCode },
    };
  },
};
