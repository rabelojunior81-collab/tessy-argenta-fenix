```typescript
// Use js-tiktoken for accurate OpenAI tokenization
// Fallback to heuristic for other providers

const MODEL_LIMITS: Record<string, number> = {
  'gpt-4o': 128_000,
  'gpt-4o-mini': 128_000,
  'gpt-4-turbo': 128_000,
  'gpt-5': 200_000,
  'claude-sonnet-4-6': 200_000,
  'claude-haiku-4-5-20251001': 200_000,
  'claude-opus-4-5-20251101': 200_000,
  'gemini-2.0-flash': 1_000_000,
  'gemini-3-pro': 1_000_000,
  'gemini-3-flash': 1_000_000,
  'qwen2.5-coder:14b': 128_000,
  'qwen2.5-coder:7b': 128_000,
  // Ollama models approximate
  'llama3': 128_000,
  'mistral': 128_000,
};

export class TokenCounter {
  // Cache tiktoken encoders per model
  private encoders = new Map<string, any>();

  async count(messages: Array<{ role: string; content: string }>, model: string): Promise<number> {
    // OpenAI models: use tiktoken
    if (model.startsWith('gpt-') || model.startsWith('o')) {
      return this.countOpenAI(messages);
    }

    // Anthropic: use their token counting API if available, else heuristic
    if (model.startsWith('claude-')) {
      return this.countHeuristic(messages, 4); // ~4 chars per token
    }

    // Gemini/Ollama: simple heuristic
    return this.countHeuristic(messages, 4);
  }

  private async countOpenAI(messages: Array<{ role: string; content: string }>): Promise<number> {
    // Lazy load tiktoken
    const tiktoken = await import('js-tiktoken');
    const encoder = await tiktoken.encodingForModel('o200k_base');
    
    let total = 0;
    for (const msg of messages) {
      total += encoder.encode(msg.content).length;
      total += 4; // per-message overhead
    }
    total += 3; // assistant overhead
    return total;
  }

  private countHeuristic(messages: Array<{ role: string; content: string }>, charsPerToken: number): number {
    return messages.reduce((sum, m) => sum + Math.ceil(m.content.length / charsPerToken), 0);
  }

  async estimate(
    messages: Array<{ role: string; content: string }>,
    model: string
  ): Promise<{ used: number; limit: number; percentage: number }> {
    const used = await this.count(messages, model);
    const limit = MODEL_LIMITS[model] ?? 100_000;
    return { used, limit, percentage: Math.min(100, (used / limit) * 100) };
  }

  getLimit(model: string): number {
    return MODEL_LIMITS[model] ?? 100_000;
  }
}
```

**IMPORTANT:** Use `js-tiktoken` for accurate OpenAI token counting. Install via `npm i js-tiktoken@1.0.21` if needed. Do NOT use simple character count for OpenAI models — this will be wrong.

```typescript
import type { ChatMessage } from '../providers/ProviderStyle';
import { TokenCounter } from './TokenCounter';

export interface CompactResult {
  compacted: ChatMessage[];
  reason: string;
  removedCount: number;
}

export class SlidingWindow {
  private systemPrompt: ChatMessage | null = null;

  /**
   * Compact messages to fit within token budget.
   * CRITICAL: System prompt is ALWAYS preserved.
   * 
   * @returns compacted message array with system prompt at start
   */
  async compact(
    messages: ChatMessage[],
    model: string,
    counter: TokenCounter,
    onCompact?: (reason: string) => void
  ): Promise<ChatMessage[]> {
    // 1. Check if compaction needed
    const { used, limit } = await counter.estimate(
      messages.map(m => ({ role: m.role, content: m.content })),
      model
    );

    if (used < limit * 0.8) {
      return messages; // Within budget — no compaction needed
    }

    // 2. Extract and PRESERVE system prompt
    const systemMsg = messages.find(m => m.role === 'system');
    if (systemMsg) {
      this.systemPrompt = systemMsg;
    }

    // 3. Keep last N messages (recent context is most valuable)
    const RECENT_COUNT = 8;
    const recent = messages.filter(m => m.role !== 'system').slice(-RECENT_COUNT);

    // 4. Compact middle messages into summary
    const middle = messages.filter(m => m.role !== 'system').slice(0, -RECENT_COUNT);
    
    if (middle.length > 0 && this.systemPrompt) {
      const summary = this.summarize(middle);
      const summaryMsg: ChatMessage = {
        role: 'system',
        content: `[Compacted] Summary of ${middle.length} earlier messages:\n${summary}`,
      };

      const compacted = [this.systemPrompt, summaryMsg, ...recent];
      
      const removedCount = middle.length;
      onCompact?.(`Compacted ${removedCount} messages — system prompt preserved`);

      return compacted;
    }

    // No middle messages or no system prompt
    return this.systemPrompt 
      ? [this.systemPrompt, ...recent]
      : recent;
  }

  /**
   * Summarize messages for compaction.
   * Uses a lightweight model or heuristic summary.
   */
  private summarize(messages: ChatMessage[]): string {
    if (messages.length === 0) return '';

    // Simple extractive summary: first line of each message
    const lines = messages.map((m, i) => {
      const firstLine = m.content.split('\n')[0].slice(0, 100);
      return `[${i + 1}] ${m.role}: ${firstLine}`;
    });

    return lines.join('\n');
  }

  /**
   * Force reset — clears system prompt cache
   */
  reset(): void {
    this.systemPrompt = null;
  }
}
```
