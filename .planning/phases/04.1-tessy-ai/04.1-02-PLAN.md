Create `src/lib/ai/providers/OpenAIStyle.ts` implementing the ProviderStyle interface for OpenAI-compatible APIs (OpenAI, OpenRouter, custom baseUrl).

**Wire format:** SSE with `data: {...}` JSON events. Uses `eventsource-parser` for stream parsing. See RESEARCH.md section 2.1 for exact format.

```typescript
import { createRequire } from 'module';
import type { ChatRequest, ChatChunk, ProviderStyle } from './ProviderStyle';

const require = createRequire(import.meta.url);

export class OpenAIStyle implements ProviderStyle {
  readonly id = 'openai' as const;
  readonly defaultModel = 'gpt-4o';
  readonly displayName = 'OpenAI';

  private baseUrl: string;
  private apiKey: string;

  constructor(config: { baseUrl?: string; apiKey: string }) {
    this.baseUrl = config.baseUrl ?? 'https://api.openai.com/v1';
    this.apiKey = config.apiKey;
  }

  async *stream(req: ChatRequest): AsyncIterable<ChatChunk> {
    const response = await fetch(`${this.baseUrl}/chat/completions`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${this.apiKey}`,
      },
      body: JSON.stringify({
        model: req.model,
        messages: req.messages.map(m => ({
          role: m.role,
          content: m.content,
          ...(m.toolCallId ? { tool_call: { id: m.toolCallId, function: { name: m.toolName, arguments: JSON.stringify(m.toolResult ?? {}) } } } : {}),
        })),
        stream: true,
        tools: req.tools?.map(t => ({
          type: 'function',
          function: { name: t.name, description: t.description, parameters: {} }, // Zod schema to JSON schema conversion deferred
        })),
        max_tokens: req.maxTokens ?? 4096,
        temperature: req.temperature ?? 0.2,
      }),
      signal: req.signal,
    });

    if (!response.ok) {
      const error = await response.text();
      throw new Error(`OpenAI API error: ${response.status} ${error}`);
    }

    if (!response.body) {
      throw new Error('No response body');
    }

    // SSE parsing: use eventsource-parser (NOT EventSource API — no POST support)
    const { createParser } = await import('eventsource-parser');
    
    const stream = response.body
      .pipeThrough(new TextDecoderStream())
      .pipeThrough(createParser(stream => {
        // eventsource-parser callback — process each event
        if (stream.data === '[DONE]') return;
        try {
          const parsed = JSON.parse(stream.data);
          const delta = parsed.choices?.[0]?.delta;
          
          if (delta?.content) {
            yield { kind: 'text', text: delta.content };
          }
          
          if (delta?.tool_calls) {
            for (const tc of delta.tool_calls) {
              yield {
                kind: 'tool_call_delta',
                toolCallId: tc.id,
                toolName: tc.function.name,
                argDelta: tc.function.arguments ?? '',
              };
            }
          }
        } catch {}
      }));

    // Read from stream — simplified implementation
    const reader = response.body.getReader();
    const decoder = new TextDecoder();
    const parser = createParser((event) => {
      if (event.data === '[DONE]') return;
      try {
        const parsed = JSON.parse(event.data);
        const delta = parsed.choices?.[0]?.delta;
        
        if (delta?.content) {
          return { kind: 'text' as const, text: delta.content };
        }
        
        if (delta?.tool_calls) {
          for (const tc of delta.tool_calls) {
            return {
              kind: 'tool_call_delta' as const,
              toolCallId: tc.id,
              toolName: tc.function.name,
              argDelta: tc.function.arguments ?? '',
            };
          }
        }
      } catch {}
    });

    while (true) {
      const { done, value } = await reader.read();
      if (done) break;
      parser.feed(decoder.decode(value));
    }
  }
}
```

**CRITICAL:** Use `eventsource-parser` (NOT EventSource API — EventSource doesn't support POST). Install via `npm i eventsource-parser@3.0.8` if needed.

**IMPORTANT:** This is a template — the actual implementation needs to properly handle the async generator protocol. See RESEARCH.md section 2.1 for the exact SSE wire format and parsing strategy.

Create `src/lib/ai/providers/AnthropicStyle.ts` implementing the ProviderStyle interface for Anthropic's API.

**Wire format:** SSE with named events (`event: content_block_delta`). Headers include `anthropic-version: 2023-06-01`. See RESEARCH.md section 2.2 for exact format.

```typescript
import type { ChatRequest, ChatChunk, ProviderStyle } from './ProviderStyle';

export class AnthropicStyle implements ProviderStyle {
  readonly id = 'anthropic' as const;
  readonly defaultModel = 'claude-sonnet-4-20250514';
  readonly displayName = 'Anthropic';

  private apiKey: string;

  constructor(config: { apiKey: string }) {
    this.apiKey = config.apiKey;
  }

  async *stream(req: ChatRequest): AsyncIterable<ChatChunk> {
    const response = await fetch('https://api.anthropic.com/v1/messages', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'x-api-key': this.apiKey,
        'anthropic-version': '2023-06-01',
      },
      body: JSON.stringify({
        model: req.model,
        messages: req.messages.map(m => ({ role: m.role, content: m.content })),
        stream: true,
        max_tokens: req.maxTokens ?? 4096,
        tools: req.tools as any, // Tool spec format differs from OpenAI
      }),
      signal: req.signal,
    });

    if (!response.ok) {
      throw new Error(`Anthropic API error: ${response.status}`);
    }

    if (!response.body) {
      throw new Error('No response body');
    }

    // Anthropic SSE: uses named events, not data: prefix
    // event: content_block_delta\ndata: {...}\n\n
    const reader = response.body.getReader();
    const decoder = new TextDecoder();
    let currentEvent = '';
    let currentData = '';

    while (true) {
      const { done, value } = await reader.read();
      if (done) break;
      
      const chunk = decoder.decode(value);
      const lines = chunk.split('\n');
      
      for (const line of lines) {
        if (line.startsWith('event: ')) {
          currentEvent = line.slice(7).trim();
        } else if (line.startsWith('data: ')) {
          currentData = line.slice(6).trim();
          
          if (currentEvent === 'content_block_delta' && currentData) {
            try {
              const parsed = JSON.parse(currentData);
              const delta = parsed.delta;
              
              if (delta?.type === 'text_delta') {
                yield { kind: 'text', text: delta.text };
              }
              
              if (delta?.type === 'input_json_delta') {
                yield {
                  kind: 'tool_call_delta',
                  toolCallId: String(parsed.index),
                  argDelta: delta.partial_json,
                };
              }
            } catch {}
          }
          
          if (currentEvent === 'message_stop') {
            yield { kind: 'done' };
          }
        }
      }
    }
  }
}
```

**CRITICAL:** Anthropic uses named SSE events (`event: content_block_delta` not `data:`). This is different from OpenAI. Use the correct parsing approach.

Create `src/lib/ai/providers/GeminiStyle.ts` implementing the ProviderStyle interface for Google's Gemini API.

**Wire format:** SSE via `?alt=sse` query param. Note: Gemini sends COMPLETE objects per chunk (not deltas) for function calls. Text comes as chunks. See RESEARCH.md section 2.3 for exact format.

```typescript
import type { ChatRequest, ChatChunk, ProviderStyle } from './ProviderStyle';

export class GeminiStyle implements ProviderStyle {
  readonly id = 'gemini' as const;
  readonly defaultModel = 'gemini-2.0-flash';
  readonly displayName = 'Google Gemini';

  private apiKey: string;

  constructor(config: { apiKey: string }) {
    this.apiKey = config.apiKey;
  }

  async *stream(req: ChatRequest): AsyncIterable<ChatChunk> {
    const model = req.model.includes('/') ? req.model : `models/${req.model}`;
    const url = `https://generativelanguage.googleapis.com/v1beta/${model}:generateContent?alt=sse`;

    const response = await fetch(url, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${this.apiKey}`,
      },
      body: JSON.stringify({
        contents: req.messages
          .filter(m => m.role !== 'system')
          .map(m => ({ role: m.role === 'assistant' ? 'model' : 'user', parts: [{ text: m.content }] })),
        systemInstruction: req.messages.find(m => m.role === 'system') 
          ? { parts: [{ text: m.messages.find(m => m.role === 'system')!.content }] }
          : undefined,
        tools: req.tools ? [{ functionDeclarations: req.tools.map(t => ({ name: t.name, description: t.description })) }] : undefined,
        generationConfig: {
          temperature: req.temperature ?? 0.2,
          maxOutputTokens: req.maxTokens ?? 4096,
        },
        stream: true,
      }),
      signal: req.signal,
    });

    if (!response.ok) {
      throw new Error(`Gemini API error: ${response.status}`);
    }

    if (!response.body) {
      throw new Error('No response body');
    }

    const reader = response.body.getReader();
    const decoder = new TextDecoder();

    while (true) {
      const { done, value } = await reader.read();
      if (done) break;
      
      const lines = decoder.decode(value).split('\n').filter(Boolean);
      
      for (const line of lines) {
        if (!line.startsWith('data: ')) continue;
        
        try {
          const parsed = JSON.parse(line.slice(5));
          const candidate = parsed.candidates?.[0];
          if (!candidate) continue;
          
          const part = candidate.content?.parts?.[0];
          
          if (part?.text) {
            yield { kind: 'text', text: part.text };
          }
          
          if (part?.functionCall) {
            // Gemini sends COMPLETE functionCall object, NOT delta
            yield {
              kind: 'tool_call_complete',
              toolCallId: part.functionCall.name, // Gemini has no ID — use name
              toolName: part.functionCall.name,
              args: part.functionCall.args,
            };
          }
        } catch {}
      }
    }
  }
}
```

**CRITICAL:** Gemini requires `?alt=sse` query param — without it, you get JSON array instead of SSE. Gemini also sends COMPLETE tool call objects per chunk (not incremental deltas). Use `kind: 'tool_call_complete'` not `tool_call_delta`.

Create `src/lib/ai/providers/OllamaStyle.ts` implementing the ProviderStyle interface for Ollama local models.

**Wire format:** NDJSON (newline-delimited JSON) — NOT SSE. Each line is a complete JSON object. The stream terminates when `done: true` appears. See RESEARCH.md section 2.4 for exact format.

```typescript
import type { ChatRequest, ChatChunk, ProviderStyle } from './ProviderStyle';

export class OllamaStyle implements ProviderStyle {
  readonly id = 'ollama' as const;
  readonly defaultModel = 'qwen2.5-coder:14b';
  readonly displayName = 'Ollama (Local)';

  private baseUrl: string;

  constructor(config: { baseUrl?: string } = {}) {
    this.baseUrl = config.baseUrl ?? 'http://localhost:11434';
  }

  async *stream(req: ChatRequest): AsyncIterable<ChatChunk> {
    const response = await fetch(`${this.baseUrl}/api/chat`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        model: req.model,
        messages: req.messages.map(m => ({ role: m.role, content: m.content })),
        stream: true,
        tools: req.tools as any,
      }),
      signal: req.signal,
    });

    if (!response.ok) {
      throw new Error(`Ollama error: ${response.status}`);
    }

    if (!response.body) {
      throw new Error('No response body');
    }

    // Ollama uses NDJSON — read lines, parse each JSON object
    const reader = response.body.getReader();
    const decoder = new TextDecoder();
    let buffer = '';

    while (true) {
      const { done, value } = await reader.read();
      if (done) break;
      
      buffer += decoder.decode(value, { stream: true });
      const lines = buffer.split('\n');
      buffer = lines.pop() ?? ''; // Keep incomplete line in buffer
      
      for (const line of lines) {
        if (!line.trim()) continue;
        
        try {
          const parsed = JSON.parse(line);
          const msg = parsed.message;
          
          if (msg?.content) {
            yield { kind: 'text', text: msg.content };
          }
          
          if (msg?.tool_calls) {
            for (const tc of msg.tool_calls) {
              yield {
                kind: 'tool_call_complete',
                toolCallId: tc.function.name, // Ollama has no ID — use name
                toolName: tc.function.name,
                args: tc.function.arguments,
              };
            }
          }
          
          if (parsed.done) {
            yield { kind: 'done' };
          }
        } catch {}
      }
    }
  }
}
```

**CRITICAL:** Ollama uses NDJSON, NOT SSE. Do NOT use eventsource-parser. Use TextDecoderStream + line-by-line JSON parsing. The `done: true` field terminates the stream — there is no `[DONE]` sentinel.

**ProviderRegistry.ts:** Factory that maps provider ID to ProviderStyle instance. Handles API key retrieval from storage.

```typescript
import type { ProviderStyle } from './ProviderStyle';
import { OpenAIStyle } from './OpenAIStyle';
import { AnthropicStyle } from './AnthropicStyle';
import { GeminiStyle } from './GeminiStyle';
import { OllamaStyle } from './OllamaStyle';
import { OpenRouterStyle } from './OpenRouterStyle';

type ProviderId = 'openai' | 'anthropic' | 'gemini' | 'ollama' | 'openrouter';

interface SecretStore {
  get(provider: string, key: string): Promise<string | null>;
}

export class ProviderRegistry {
  private providers = new Map<ProviderId, ProviderStyle>();
  private secrets: SecretStore;

  constructor(secrets: SecretStore) {
    this.secrets = secrets;
  }

  async get(providerId: ProviderId): Promise<ProviderStyle> {
    if (this.providers.has(providerId)) {
      return this.providers.get(providerId)!;
    }

    const apiKey = await this.secrets.get(providerId, 'api_key');
    if (!apiKey) {
      throw new Error(`No API key for provider: ${providerId}`);
    }

    let provider: ProviderStyle;
    switch (providerId) {
      case 'openai':
        provider = new OpenAIStyle({ apiKey });
        break;
      case 'anthropic':
        provider = new AnthropicStyle({ apiKey });
        break;
      case 'gemini':
        provider = new GeminiStyle({ apiKey });
        break;
      case 'ollama':
        provider = new OllamaStyle();
        break;
      case 'openrouter':
        provider = new OpenRouterStyle({ apiKey });
        break;
      default:
        throw new Error(`Unknown provider: ${providerId}`);
    }

    this.providers.set(providerId, provider);
    return provider;
  }
}
```

**IMPORTANT:** The `SecretStore` interface is the same pattern used in ToolBroker for secret hydration. This ensures consistent credential management across the AI system.
